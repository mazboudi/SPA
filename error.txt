".windows_build":
  stage: build
  tags:
  - Windows
  variables:
    PSADT_FRAMEWORK_VERSION: 4.1.0
  script:
  - |
    $ErrorActionPreference = "Stop"
    $POWERSHELL_TELEMETRY_OPTOUT = "1"

    Write-Host "=== Windows Build: $($env:CI_PROJECT_NAME) ===" -ForegroundColor Cyan
    Write-Host "PSADT version  : $($env:PSADT_FRAMEWORK_VERSION)"
    Write-Host "Pipeline tag   : $($env:CI_COMMIT_TAG)"

    ## 1 — Read and validate app metadata
    if (!(Test-Path 'app.json')) { throw "app.json not found" }
    $app = Get-Content 'app.json' -Raw | ConvertFrom-Json

    Write-Host "--- Validating app.json ---"
    foreach ($f in @('title','publisher','package_id','version','platforms','deployment')) {
      if (-not $app.$f) { throw "app.json: missing required field '$f'" }
    }
    if ($app.lifecycle -eq 'retired') {
      Write-Warning "app.json: lifecycle=retired — skipping build."
      exit 0
    }
    Write-Host "  ✓ app.json valid"

    if (!(Test-Path 'windows/package.yaml')) { throw "windows/package.yaml not found" }

    Write-Host "--- Validating windows/package.yaml ---"
    $winPkg = Get-Content 'windows/package.yaml' -Raw
    foreach ($f in @('vendor_version','packaging_version','installer_type','install','uninstall','detection')) {
      if ($winPkg -notmatch "(?m)^${f}:") { throw "windows/package.yaml: missing required field '$f'" }
    }
    if ($winPkg -match '(?m)^installer_type:\s*(\S+)') {
      $iType = $Matches[1].Trim(" '`"")
      if ($iType -notin @('msi','exe','msix','ps1')) {
        throw "windows/package.yaml: installer_type '$iType' must be one of: msi, exe, msix, ps1"
      }
    }
    Write-Host "  ✓ windows/package.yaml valid"

    ## 2 — Download psadt-enterprise framework bundle (includes IntuneWinAppUtil.exe)
    $frameworkDir = "psadt-framework-$($env:PSADT_FRAMEWORK_VERSION)"
    if (!(Test-Path $frameworkDir)) {
      $pkgUri = "$($env:CI_API_V4_URL)/projects/$($env:PSADT_PROJECT_ID)/packages/generic/psadt-enterprise/$($env:PSADT_FRAMEWORK_VERSION)/psadt-enterprise-$($env:PSADT_FRAMEWORK_VERSION).zip"
      Write-Host "Downloading PSADT bundle from: $pkgUri"
      $headers = @{ 'PRIVATE-TOKEN' = $env:GITLAB_READ_TOKEN }
      $zipPath = "$frameworkDir.zip"
      Invoke-WebRequest -Uri $pkgUri -Headers $headers -OutFile $zipPath
      Expand-Archive -Path $zipPath -DestinationPath $frameworkDir -Force
      Remove-Item $zipPath
      Write-Host "Framework extracted to: $frameworkDir"
    }

    ## 3 — Resolve IntuneWinAppUtil.exe
    ##     Priority: C:\tools (pre-installed on runner) → bundle tools\ → bundle root
    $intuneUtil = $null
    $candidates = @(
      'C:\tools\IntuneWinAppUtil.exe',                          # pre-installed on runner
      (Join-Path $frameworkDir 'tools\IntuneWinAppUtil.exe'),   # bundled in psadt-enterprise zip
      (Join-Path $frameworkDir 'IntuneWinAppUtil.exe')           # legacy bundle root location
    )
    foreach ($candidate in $candidates) {
      if (Test-Path $candidate) { $intuneUtil = $candidate; break }
    }
    if (-not $intuneUtil) {
      throw "IntuneWinAppUtil.exe not found. Install it at C:\tools\IntuneWinAppUtil.exe on the runner, or include tools/ in the psadt-enterprise bundle."
    }
    Write-Host "Using IntuneWinAppUtil: $intuneUtil"


    ##     In production, the psadt-enterprise bundle should already include
    ##     the upstream PSADT in a PSAppDeployToolkit/ subfolder.
    ##     If not present, we pull it from the GitLab Package Registry or a
    ##     known URL. This step is a placeholder — update the URL for your env.
    $psadtRuntime = Join-Path $frameworkDir 'PSAppDeployToolkit'
    if (!(Test-Path $psadtRuntime)) {
      Write-Warning "PSAppDeployToolkit runtime not found in framework bundle."
      Write-Warning "Expected path: $psadtRuntime"
      Write-Warning "Please add the PSADT runtime to the psadt-enterprise bundle."
    }

    ## 5 — Build staging area
    $staging = Join-Path $env:TEMP ("psadt_build_" + [guid]::NewGuid())
    New-Item -ItemType Directory -Path $staging | Out-Null

    # Copy enterprise framework template
    Copy-Item -Path (Join-Path $frameworkDir '*') -Destination $staging -Recurse -Force

    # Overlay title's Deploy-Application.ps1 if present
    $titleDeploy = 'windows/src/Deploy-Application.ps1'
    if (Test-Path $titleDeploy) {
      Write-Host "Overlaying title Deploy-Application.ps1..."
      Copy-Item -Path $titleDeploy -Destination $staging -Force
    }

    # Copy installer from runner VM if WINDOWS_INSTALLER_SOURCE is set
    # Use this when the installer is pre-staged on the runner (e.g. C:\files\7-zip\7z2408-x64.msi)
    # rather than committed to git in windows/src/Files/
    if ($env:WINDOWS_INSTALLER_SOURCE) {
      $filesDir = 'windows\src\Files'
      if (!(Test-Path $filesDir)) { New-Item -ItemType Directory -Path $filesDir | Out-Null }
      if (Test-Path $env:WINDOWS_INSTALLER_SOURCE) {
        Write-Host "Copying installer from: $($env:WINDOWS_INSTALLER_SOURCE) -> $filesDir"
        Copy-Item -Path $env:WINDOWS_INSTALLER_SOURCE -Destination $filesDir -Force
      } else {
        throw "WINDOWS_INSTALLER_SOURCE set but file not found: $($env:WINDOWS_INSTALLER_SOURCE)"
      }
    }

    # Overlay src/ content (Files, SupportFiles, Assets)
    foreach ($subDir in @('Files','SupportFiles','Assets')) {
      $srcPath = Join-Path 'windows/src' $subDir
      if (Test-Path $srcPath) {
        $dstPath = Join-Path $staging $subDir
        if (!(Test-Path $dstPath)) { New-Item -ItemType Directory -Path $dstPath | Out-Null }
        Copy-Item -Path (Join-Path $srcPath '*') -Destination $dstPath -Recurse -Force
      }
    }

    # Overlay Config and Strings overrides
    foreach ($cfgFile in @('Config\config.psd1','Strings\strings.psd1')) {
      $src = Join-Path 'windows/src' $cfgFile
      if (Test-Path $src) {
        $dst = Join-Path $staging $cfgFile
        New-Item -ItemType Directory -Force -Path (Split-Path $dst) | Out-Null
        Copy-Item -Path $src -Destination $dst -Force
      }
    }

    ## 6 — Run IntuneWinAppUtil
    $outDir = 'out'
    if (!(Test-Path $outDir)) { New-Item -ItemType Directory -Path $outDir | Out-Null }

    $setupFile = 'Invoke-AppDeployToolkit.exe'
    $setupPath = Join-Path $staging $setupFile
    if (!(Test-Path $setupPath)) {
      # Fallback: look for Deploy-Application.ps1 as entry point
      $setupFile = 'Deploy-Application.ps1'
    }

    Write-Host "Running IntuneWinAppUtil..."
    & $intuneUtil -c $staging -s $setupFile -o (Resolve-Path $outDir) -q

    ## 7 — Rename output deterministically
    $safeName    = ($app.package_id -replace '[^a-z0-9\-]', '-')
    $winPkg      = Get-Content 'windows/package.yaml' -Raw

    # Extract vendor_version with simple regex
    if ($winPkg -match 'vendor_version:\s*["''']?([^\n"'''\r]+)') {
      $vendorVer = $Matches[1].Trim()
    } else { $vendorVer = $app.version }

    if ($winPkg -match 'packaging_version:\s*(\d+)') {
      $pkgVer = $Matches[1].Trim()
    } else { $pkgVer = '1' }

    $produced = Get-ChildItem -Path $outDir -Filter '*.intunewin' | Sort-Object LastWriteTime -Descending | Select-Object -First 1
    if (-not $produced) { throw "No .intunewin file produced in $outDir" }

    $targetName = "${safeName}_${vendorVer}_${pkgVer}.intunewin"
    $targetPath = Join-Path $outDir $targetName
    Move-Item -Path $produced.FullName -Destination $targetPath -Force

    Write-Host "✅ Built: $targetPath" -ForegroundColor Green

    ## 8 — Compute checksum and write dotenv
    $sha256 = (Get-FileHash -Path $targetPath -Algorithm SHA256).Hash.ToLower()
    $builtAt = (Get-Date -Format 'o')

    $dotenv = @"
    INTUNEWIN_PATH=$targetPath
    INTUNEWIN_FILENAME=$targetName
    INTUNEWIN_SHA256=$sha256
    PACKAGE_ID=$($app.package_id)
    VENDOR_VERSION=$vendorVer
    PACKAGING_VERSION=$pkgVer
    PLATFORM=windows
    BUILT_AT=$builtAt
    "@
          $dotenv | Out-File (Join-Path $outDir 'build.env') -Encoding ascii -Force
          Write-Host "Dotenv written: out/build.env"

          ## 9 — Cleanup staging
          Remove-Item -Path $staging -Recurse -Force -ErrorAction SilentlyContinue
  artifacts:
    when: always
    expire_in: 14 days
    paths:
    - out/*.intunewin
    - out/build.env
    reports:
      dotenv: out/build.env

".download_intune_modules": |
  $ErrorActionPreference = "Stop"
  $modulesDir = 'intune-modules'
  if (!(Test-Path $modulesDir)) {
    New-Item -ItemType Directory -Path $modulesDir | Out-Null
  }
  $headers = @{ 'PRIVATE-TOKEN' = $env:GITLAB_READ_TOKEN }
  $scripts = @(
    'Publish-Win32App.ps1',
    'Update-Win32App.ps1',
    'Resolve-DetectionRules.ps1',
    'Resolve-Requirements.ps1',
    'Set-Win32Assignments.ps1',
    'Set-Win32Supersedence.ps1',
    'IntuneDeployment.psm1'
  )
  $ref = $env:INTUNE_MODULES_REF ?? 'main'
  foreach ($s in $scripts) {
    $encoded = [uri]::EscapeDataString("scripts/$s")
    $uri = "$($env:CI_API_V4_URL)/projects/$($env:INTUNE_MODULES_PROJECT_ID)/repository/files/$encoded/raw?ref=$ref"
    $dest = Join-Path $modulesDir $s
    try {
      Invoke-WebRequest -Uri $uri -Headers $headers -OutFile $dest -ErrorAction Stop
    } catch {
      Write-Warning "Could not download $s from intune-deployment-modules. Skipping."
    }
  }
  Write-Host "Intune modules ready in: $modulesDir"

windows_publish_intune:
  stage: publish
  tags:
  - Windows
  variables:
    INTUNE_DRY_RUN: 'false'
    INTUNE_MODULES_REF: main
  needs:
  - job: windows_build
    artifacts: true
    optional: true
  script:
  - |
    $ErrorActionPreference = "Stop"
    $modulesDir = 'intune-modules'
    if (!(Test-Path $modulesDir)) {
      New-Item -ItemType Directory -Path $modulesDir | Out-Null
    }
    $headers = @{ 'PRIVATE-TOKEN' = $env:GITLAB_READ_TOKEN }
    $scripts = @(
      'Publish-Win32App.ps1',
      'Update-Win32App.ps1',
      'Resolve-DetectionRules.ps1',
      'Resolve-Requirements.ps1',
      'Set-Win32Assignments.ps1',
      'Set-Win32Supersedence.ps1',
      'IntuneDeployment.psm1'
    )
    $ref = $env:INTUNE_MODULES_REF ?? 'main'
    foreach ($s in $scripts) {
      $encoded = [uri]::EscapeDataString("scripts/$s")
      $uri = "$($env:CI_API_V4_URL)/projects/$($env:INTUNE_MODULES_PROJECT_ID)/repository/files/$encoded/raw?ref=$ref"
      $dest = Join-Path $modulesDir $s
      try {
        Invoke-WebRequest -Uri $uri -Headers $headers -OutFile $dest -ErrorAction Stop
      } catch {
        Write-Warning "Could not download $s from intune-deployment-modules. Skipping."
      }
    }
    Write-Host "Intune modules ready in: $modulesDir"
  - |
    $ErrorActionPreference = "Stop"
    if (-not $env:INTUNEWIN_PATH) { throw "INTUNEWIN_PATH not set. Check build.env." }

    # Import shared module
    if (Test-Path 'intune-modules/IntuneDeployment.psm1') {
      Import-Module (Resolve-Path 'intune-modules/IntuneDeployment.psm1') -Force
    }

    # Build detection rules from windows/package.yaml
    $detectionRules = & 'intune-modules/Resolve-DetectionRules.ps1' `
      -PackageYamlPath 'windows/package.yaml'

    # Build requirement rules from windows/intune/requirements.json
    $requirementRules = $null
    if (Test-Path 'windows/intune/requirements.json') {
      $requirementRules = & 'intune-modules/Resolve-Requirements.ps1' `
        -RequirementsPath 'windows/intune/requirements.json'
    }

    # Publish (creates or updates) the Win32 app
    & 'intune-modules/Publish-Win32App.ps1' `
      -IntuneWinPath      $env:INTUNEWIN_PATH `
      -AppJsonPath        'windows/intune/app.json' `
      -PackageYamlPath    'windows/package.yaml' `
      -DetectionRules     $detectionRules `
      -RequirementRules   $requirementRules `
      -TenantId           $env:AZURE_TENANT_ID `
      -ClientId           $env:AZURE_CLIENT_ID `
      -ClientSecret       $env:AZURE_CLIENT_SECRET `
      -DryRun:([bool]::Parse($env:INTUNE_DRY_RUN))
  artifacts:
    when: always
    expire_in: 14 days
    paths:
    - out/app.env
    - out/publish-logs/*
    reports:
      dotenv:
      - out/app.env
  extends: ".windows_publish_intune"
  rules:
  - if: $CI_COMMIT_TAG =~ /^v\d+\.\d+\.\d+$/ && $WINDOWS_ENABLED == "true"
